1. git init - створити репозиторій
2. git clone назва репозиторію - скачати проект з GitHub
3. git add . - зафіксувати зміни
4. git commit -m 'Hello' - зробити коміт
5. git push - запушити до GitHub
6. git push --set-upstream назва_гілки - додати гілку до репозиторію
7. git s - перевірити статус фіксації add
8. git g - перевірити коміти, дату створення і гілки
9. git co -b vetka - створити гілку і одразу бути в ній
10. git br vetka - створити гілку
11. git br - перевірити гілки які існують
12. git br -v - перевірити на якій гілки та останній коміт
13. git br -m назва_гілки - змінити назву гілки
14. git switch vetka або git co vetka - переключити гілку
15. git co . або git co назва_файла - отмєна до НЕ фіксації
16. git reset название_файла або git reset . - отмєна до add
17. git co -d - видалити гілку
18. git co код_коміта - перейти до коміту
19. git log -5 - вивести останні 5 комітів
20. git reset --hard HEAD^1 - відмінити коміт до останнього 
21. git reset --soft HEAD^1 - відмінити коміт але змінити залишити 
22. git pull origin назва_гілки - відтягнути зміни
23. git commit --amend -m "назва коміту" - змінити коміт содєржиме
24.git merge назва_гілки - слити гілки
